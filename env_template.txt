# Создайте файл config/.env со следующим содержимым:

# Google Gemini API Key
GOOGLE_API_KEY=YOUR_GEMINI_API_KEY_HERE

# Telegram Bot Token
TELEGRAM_BOT_TOKEN=YOUR_TELEGRAM_BOT_TOKEN_HERE

# YouTube API (опционально)
YOUTUBE_CLIENT_SECRET_FILE=config/client_secret.json

# Инструкция:
# 1. Скопируйте содержимое выше
# 2. Создайте файл config/.env
# 3. Вставьте содержимое в файл
# 4. Замените YOUR_*_HERE на ваши реальные ключи
# 5. Сохраните файл
# 6. НИКОГДА не коммитьте файл .env в Git!
